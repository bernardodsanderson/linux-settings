## Command Line

sudo apt-get update
sudo apt-get upgrade

sudo apt-get install vim, git, transmission, filezilla, vlc, steam, zsh

### Node
curl -sL https://deb.nodesource.com/setup_4.x | sudo -E bash -
sudo apt-get install -y nodejs

### ZSH
zsh --version
chsh -s $(which zsh)
npm install --global pure-prompt
vim ~/.zshrc
add > ZSH_THEME="pure"
